39constexpr void validate_ascii_char([[maybe_unused]] char c) noexcept { gsl_Expects((c & 0x80) == 0); }

41template<std::size_t N>
42constexpr void validate_ascii_string([[maybe_unused]] const char (&s)[N + 1]) noexcept
43{
44#ifndef NDEBUG
45 if constexpr (N != 0)
46 for (size_t i = 0; i < N; ++i) validate_ascii_char(s[i]);
47#endif
48}

64template<typename StandardCharT, std::size_t N, std::size_t M>

147 [[nodiscard]] friend constexpr auto operator<=>(const basic_symbol_text& lhs,
148 const basic_symbol_text<StandardCharT2, N2, M2>& rhs) noexcept
149 {
150 UNITS_DIAGNOSTIC_PUSH
151 UNITS_DIAGNOSTIC_IGNORE_ZERO_AS_NULLPOINTER_CONSTANT
152 if (const auto cmp = lhs.standard() <=> rhs.standard(); cmp != 0) return cmp;
153 UNITS_DIAGNOSTIC_POP
154 return lhs.ascii() <=> rhs.ascii();

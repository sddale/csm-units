32struct base_dimension {
33 const char* name;
34};
35
36constexpr bool operator==(const base_dimension& lhs, const base_dimension& rhs)
37{
38 const char* p1 = lhs.name;
39 const char* p2 = rhs.name;
40 for (; (*p1 != '\0') && (*p2 != '\0'); ++p1, (void)++p2) {
41 if (*p1 != *p2) return false;
42 }
43 return *p1 == *p2;
44}
45
46constexpr bool operator<(const base_dimension& lhs, const base_dimension& rhs)
47{
48 const char* p1 = lhs.name;
49 const char* p2 = rhs.name;
50 for (; (*p1 != '\0') && (*p2 != '\0'); ++p1, (void)++p2) {
51 if (*p1 < *p2) return true;
52 if (*p2 < *p1) return false;
53 }
54 return (*p1 == '\0') && (*p2 != '\0');
55}

116template<>

118 using type = dimension<>;

121template<typename E>
122struct dim_consolidate<dimension<E>> {
123 using type = dimension<E>;
124};
125
126template<typename E1, typename... ERest>
127struct dim_consolidate<dimension<E1, ERest...>> {
128 using rest = dim_consolidate_t<dimension<ERest...>>;
129 using type = conditional<is_same_v<rest, dimension<>>, dimension<E1>, type_list_push_front<rest, E1>>;
130};
131
132template<const base_dimension& D, std::intmax_t Num1, std::intmax_t Den1, std::intmax_t Num2, std::intmax_t Den2,
133 typename... ERest>
134struct dim_consolidate<dimension<exponent<D, Num1, Den1>, exponent<D, Num2, Den2>, ERest...>> {
135 using r1 = std::ratio<Num1, Den1>;
136 using r2 = std::ratio<Num2, Den2>;
137 using r = std::ratio_add<r1, r2>;
138 using type = conditional<r::num == 0, dim_consolidate_t<dimension<ERest...>>,
139 dim_consolidate_t<dimension<exponent<D, r::num, r::den>, ERest...>>>;
140};

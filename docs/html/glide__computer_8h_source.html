45// An example of a really simplified tactical glide computer
46// Simplifications:
47// - glider 100% clean and with full factory performance (brand new painting)
48// - no influence of the ballast (pilot weight, water, etc) to glider performance
49// - only one point on a glider polar curve
50// - no influence of bank angle (during circling) on a glider performance
51// - no wind
52// - constant thermals strength
53// - thermals exactly where and when we need them ;-)
54// - no airspaces
55// - ground level changes linearly between waypoints
56// - no ground obstacles (i.e. mountains) to pass
57// - flight path exactly on a shortest possible line to destination
58

70template<units::QuantityKind QK1, units::QuantityKind QK2>
71constexpr units::Dimensionless auto operator/(const QK1& lhs, const QK2& rhs)
72 requires(!units::QuantityKindRelatedTo<QK1, QK2>) && requires { lhs.common() / rhs.common(); }
73{

178 }

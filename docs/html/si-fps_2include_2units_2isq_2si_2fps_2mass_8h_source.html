37// https://en.wikipedia.org/wiki/Pound_(mass)
38struct pound : named_scaled_unit<pound, "lb", mag<ratio(45'359'237, 100'000'000)>(), si::kilogram> {};
39
40struct dim_mass : isq::dim_mass<pound> {};
41
42template<UnitOf<dim_mass> U, Representation Rep = double>
43using mass = quantity<dim_mass, U, Rep>;

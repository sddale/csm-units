99 template <typename T>
100 FMT_CONSTEXPR scan_arg(T& value) : type(scan_type::custom_type) {
101 custom.value = &value;
102 custom.scan = scan_custom_arg<T>;
103 }
104
105 private:
106 template <typename T>
107 static void scan_custom_arg(void* arg, scan_parse_context& parse_ctx,
108 scan_context& ctx) {
109 scanner<T> s;
110 parse_ctx.advance_to(s.parse(parse_ctx));
111 ctx.advance_to(s.scan(*static_cast<T*>(arg), ctx));
112 }

129struct scan_handler : error_handler {
130 private:
131 scan_parse_context parse_ctx_;
132 scan_context scan_ctx_;
133 scan_args args_;
134 int next_arg_id_;
135 scan_arg arg_;
136
137 template <typename T = unsigned> T read_uint() {
138 T value = 0;
139 auto it = scan_ctx_.begin(), end = scan_ctx_.end();
140 while (it != end) {
141 char c = *it++;
142 if (c < '0' || c > '9') on_error("invalid input");
143 // TODO: check overflow
144 value = value * 10 + static_cast<unsigned>(c - '0');
145 }
146 scan_ctx_.advance_to(it);
147 return value;
148 }
149
150 template <typename T = int> T read_int() {
151 auto it = scan_ctx_.begin(), end = scan_ctx_.end();
152 bool negative = it != end && *it == '-';
153 if (negative) ++it;
154 scan_ctx_.advance_to(it);
155 const auto value = read_uint<typename std::make_unsigned<T>::type>();
156 if (negative) return -static_cast<T>(value);
157 return static_cast<T>(value);
158 }
159
160 public:
161 FMT_CONSTEXPR scan_handler(string_view format, string_view input,
162 scan_args args)
163 : parse_ctx_(format), scan_ctx_(input), args_(args), next_arg_id_(0) {}
164
165 const char* pos() const { return scan_ctx_.begin(); }
166
167 void on_text(const char* begin, const char* end) {
168 auto size = to_unsigned(end - begin);
169 auto it = scan_ctx_.begin();
170 if (it + size > scan_ctx_.end() ||
171 !std::equal(begin, end, make_checked(it, size))) {
172 on_error("invalid input");
173 }
174 scan_ctx_.advance_to(it + size);
175 }
176
177 FMT_CONSTEXPR int on_arg_id() { return on_arg_id(next_arg_id_++); }
178 FMT_CONSTEXPR int on_arg_id(int id) {
179 if (id >= args_.size) on_error("argument index out of range");
180 arg_ = args_.data[id];
181 return id;
182 }
183 FMT_CONSTEXPR int on_arg_id(string_view id) {
184 if (id.data()) on_error("invalid format");
185 return 0;
186 }
187
188 void on_replacement_field(int, const char*) {
189 auto it = scan_ctx_.begin(), end = scan_ctx_.end();
190 switch (arg_.type) {
191 case scan_type::int_type:
192 *arg_.int_value = read_int();
193 break;
194 case scan_type::uint_type:
195 *arg_.uint_value = read_uint();
196 break;
197 case scan_type::long_long_type:
198 *arg_.long_long_value = read_int<long long>();
199 break;
200 case scan_type::ulong_long_type:
201 *arg_.ulong_long_value = read_uint<unsigned long long>();
202 break;
203 case scan_type::string_type:
204 while (it != end && *it != ' ') arg_.string->push_back(*it++);
205 scan_ctx_.advance_to(it);
206 break;
207 case scan_type::string_view_type: {
208 auto s = it;
209 while (it != end && *it != ' ') ++it;
210 *arg_.string_view = fmt::string_view(s, to_unsigned(it - s));
211 scan_ctx_.advance_to(it);
212 break;
213 }
214 case scan_type::none_type:
215 case scan_type::custom_type:
216 assert(false);
217 }
218 }
219
220 const char* on_format_specs(int, const char* begin, const char*) {
221 if (arg_.type != scan_type::custom_type) return begin;
222 parse_ctx_.advance_to(begin);
223 arg_.custom.scan(arg_.custom.value, parse_ctx_, scan_ctx_);
224 return parse_ctx_.begin();
225 }
226};

234string_view::iterator vscan(string_view input, string_view format_str,
235 scan_args args) {
236 detail::scan_handler h(format_str, input, args);
237 detail::parse_format_string<false>(format_str, h);
238 return input.begin() + (h.pos() - &*input.begin());
239}

241template <typename... Args>
242string_view::iterator scan(string_view input, string_view format_str,
243 Args&... args) {
244 return vscan(input, format_str, make_scan_args(args...));
245}

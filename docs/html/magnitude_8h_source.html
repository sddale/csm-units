146// Raise an arbitrary arithmetic type to a positive integer power at compile time.
147template<typename T>
148constexpr T int_power(T base, std::integral auto exp)
149{
150 // As this function should only be called at compile time, the terminations herein function as
151 // "parameter-compatible static_asserts", and should not result in terminations at runtime.
152 if (exp < 0) {
153 std::terminate(); // int_power only supports positive integer powers
154 }
155
156 constexpr auto checked_multiply = [](auto a, auto b) {
157 const auto result = a * b;
158 UNITS_DIAGNOSTIC_PUSH
159 UNITS_DIAGNOSTIC_IGNORE_FLOAT_EQUAL
160 if (result / a != b) {
161 std::terminate(); // Wraparound detected
162 }
163 UNITS_DIAGNOSTIC_POP
164 return result;
165 };
166
167 constexpr auto checked_square = [checked_multiply](auto a) { return checked_multiply(a, a); };
168
169 // TODO(chogg): Unify this implementation with the one in pow.h. That one takes its exponent as a
170 // template parameter, rather than a function parameter.
171
172 if (exp == 0) {
173 return T{1};
174 }
175
176 if (exp % 2 == 1) {
177 return checked_multiply(base, int_power(base, exp - 1));
178 }
179
180 return checked_square(int_power(base, exp / 2));
181}

184template<typename T>
185constexpr widen_t<T> compute_base_power(BasePower auto bp)
186{
187 // This utility can only handle integer powers. To compute rational powers at compile time, we'll
188 // need to write a custom function.
189 //
190 // Note that since this function should only be called at compile time, the point of these
191 // terminations is to act as "static_assert substitutes", not to actually terminate at runtime.
192 if (bp.power.den != 1) {
193 std::terminate(); // Rational powers not yet supported
194 }
195
196 if (bp.power.num < 0) {
197 if constexpr (std::is_integral_v<T>) {
198 std::terminate(); // Cannot represent reciprocal as integer
199 } else {
200 return T{1} / compute_base_power<T>(inverse(bp));
201 }
202 }
203
204 auto power = bp.power.num;
205 return int_power(static_cast<widen_t<T>>(bp.get_base()), power);
206}

208// A converter for the value member variable of magnitude (below).
209//
210// The input is the desired result, but in a (wider) intermediate type. The point of this function
211// is to cast to the desired type, but avoid overflow in doing so.
212template<typename To, typename From>
213// TODO(chogg): Migrate this to use `treat_as_floating_point`.
214 requires(!std::is_integral_v<To> || std::is_integral_v<From>)
215constexpr To checked_static_cast(From x)
216{
217 // This function should only ever be called at compile time. The purpose of these terminations is
218 // to produce compiler errors, because we cannot `static_assert` on function arguments.
219 if constexpr (std::is_integral_v<To>) {
220 if (!std::in_range<To>(x)) {
221 std::terminate(); // Cannot represent magnitude in this type
222 }
223 }

225 return static_cast<To>(x);

263// Undefined unless base > 1, pow >= 0, and (base ^ pow) evenly divides n.
264constexpr std::intmax_t remove_power(std::intmax_t base, std::intmax_t pow, std::intmax_t n)
265{

267 n /= base;

275constexpr bool is_valid_base_power(const BasePower auto& bp)
276{
277 if (bp.power == 0) {
278 return false;
279 }
280
281 if constexpr (std::is_same_v<decltype(bp.get_base()), std::intmax_t>) {
282 // Some prime numbers are so big, that we can't check their primality without exhausting limits on constexpr steps
283 // and/or iterations. We can still _perform_ the factorization for these by using the `known_first_factor`
284 // workaround. However, we can't _check_ that they are prime, because this workaround depends on the input being
285 // usable in a constexpr expression. This is true for `prime_factorization` (below), where the input `N` is a
286 // template parameter, but is not true for our case, where the input `bp.get_base()` is a function parameter. (See
287 // http://www.open-std.org/jtc1/sc22/wg21/docs/papers/2019/p1045r1.html for some background reading on this
288 // distinction.)
289 //
290 // In our case: we simply give up on excluding every possible ill-formed base power, and settle for catching the
291 // most likely and common mistakes.
292 if (const bool too_big_to_check = (bp.get_base() > 1'000'000'000)) {
293 return true;
294 }
295
296 return is_prime(bp.get_base());
297 } else {
298 return bp.get_base() > 0;
299 }
300}

302// A function object to apply a predicate to all consecutive pairs of values in a sequence.
303template<typename Predicate>
304struct pairwise_all {
305 Predicate predicate;
306
307 template<typename... Ts>
308 constexpr bool operator()(Ts&&... ts) const
309 {
310 // Carefully handle different sizes, avoiding unsigned integer underflow.
311 constexpr auto num_comparisons = [](auto num_elements) {
312 return (num_elements > 1) ? (num_elements - 1) : 0;
313 }(sizeof...(Ts));
314
315 // Compare zero or more pairs of neighbours as needed.
316 return [this]<std::size_t... Is>(std::tuple<Ts...>&& t, std::index_sequence<Is...>) {
317 return (predicate(std::get<Is>(t), std::get<Is + 1>(t)) && ...);
318 }(std::make_tuple(std::forward<Ts>(ts)...), std::make_index_sequence<num_comparisons>());
319 }
320};
321
322// Deduction guide: permit constructions such as `pairwise_all{std::less{}}`.
323template<typename T>
324pairwise_all(T) -> pairwise_all<T>;

326// Check whether a sequence of (possibly heterogeneously typed) values are strictly increasing.
327template<typename... Ts>
328 requires(std::is_signed_v<Ts> && ...)
329constexpr bool strictly_increasing(Ts&&... ts)
330{
331 return pairwise_all{std::less{}}(std::forward<Ts>(ts)...);
332}

453// Recursive case for the product of any two non-identity Magnitudes.
454template<auto H1, auto... T1, auto H2, auto... T2>
455constexpr Magnitude auto operator*(magnitude<H1, T1...>, magnitude<H2, T2...>)
456{
457 // Case for when H1 has the smaller base.
458 if constexpr (H1.get_base() < H2.get_base()) {
459 if constexpr (sizeof...(T1) == 0) {
460 // Shortcut for the "pure prepend" case, which makes it easier to implement some of the other cases.
461 return magnitude<H1, H2, T2...>{};
462 } else {
463 return magnitude<H1>{} * (magnitude<T1...>{} * magnitude<H2, T2...>{});
464 }
465 }
466
467 // Case for when H2 has the smaller base.
468 if constexpr (H1.get_base() > H2.get_base()) {
469 return magnitude<H2>{} * (magnitude<H1, T1...>{} * magnitude<T2...>{});
470 }
471
472 // "Same leading base" case.
473 if constexpr (H1.get_base() == H2.get_base()) {
474 constexpr auto partial_product = magnitude<T1...>{} * magnitude<T2...>{};
475
476 // Make a new base_power with the common base of H1 and H2, whose power is their powers' sum.
477 constexpr auto new_head = [&](auto head) {
478 head.power = H1.power + H2.power;
479 return head;
480 }(H1);
481
482 if constexpr (new_head.power == 0) {
483 return partial_product;
484 } else {
485 return magnitude<new_head>{} * partial_product;
486 }
487 }
488}

500// The largest integer which can be extracted from any magnitude with only a single basis vector.
501template<auto BP>
502constexpr auto integer_part(magnitude<BP>)
503{
504 constexpr auto power_num = BP.power.num;
505 constexpr auto power_den = BP.power.den;
506
507 if constexpr (std::is_integral_v<decltype(BP.get_base())> && (power_num >= power_den)) {
508 constexpr auto largest_integer_power = [=](BasePower auto bp) {
509 bp.power = (power_num / power_den); // Note: integer division intended.
510 return bp;
511 }(BP); // Note: lambda is immediately invoked.
512
513 return magnitude<largest_integer_power>{};
514 } else {
515 return magnitude<>{};
516 }
517}

541// Common Magnitude.
542//
543// The "common Magnitude" C, of two Magnitudes M1 and M2, is the largest Magnitude such that each of its inputs is
544// expressible by only positive basis powers relative to C. That is, both (M1 / C) and (M2 / C) contain only positive
545// powers in the expansion on our basis.
546//
547// For rational Magnitudes (or, more precisely, Magnitudes that are rational _relative to each other_), this reduces to
548// the familiar convention from the std::chrono library: it is the largest Magnitude C such that each input Magnitude is
549// an _integer multiple_ of C. The connection can be seen by considering the definition in the above paragraph, and
550// recognizing that both the bases and the powers are all integers for rational Magnitudes.
551//
552// For relatively _irrational_ Magnitudes (whether from irrational bases, or fractional powers of integer bases), the
553// notion of a "common type" becomes less important, because there is no way to preserve pure integer multiplication.
554// When we go to retrieve our value, we'll be stuck with a floating point approximation no matter what choice we make.
555// Thus, we make the _simplest_ choice which reproduces the correct convention in the rational case: namely, taking the
556// minimum power for each base (where absent bases implicitly have a power of 0).
557
558namespace detail {
559template<auto BP>
560constexpr auto remove_positive_power(magnitude<BP> m)
561{
562 if constexpr (BP.power.num < 0) {
563 return m;
564 } else {
565 return magnitude<>{};
566 }
567}
568
569template<auto... BPs>
570constexpr auto remove_positive_powers(magnitude<BPs...>)
571{
572 return (magnitude<>{} * ... * remove_positive_power(magnitude<BPs>{}));
573}
574} // namespace detail

578constexpr auto common_magnitude(magnitude<>, Magnitude auto m) { return detail::remove_positive_powers(m); }

581// Recursive case for the common Magnitude of any two non-identity Magnitudes.
582template<auto H1, auto... T1, auto H2, auto... T2>
583constexpr auto common_magnitude(magnitude<H1, T1...>, magnitude<H2, T2...>)
584{
585 using detail::remove_positive_power;
586
587 if constexpr (H1.get_base() < H2.get_base()) {
588 // When H1 has the smaller base, prepend to result from recursion.
589 return remove_positive_power(magnitude<H1>{}) * common_magnitude(magnitude<T1...>{}, magnitude<H2, T2...>{});
590 } else if constexpr (H2.get_base() < H1.get_base()) {
591 // When H2 has the smaller base, prepend to result from recursion.
592 return remove_positive_power(magnitude<H2>{}) * common_magnitude(magnitude<H1, T1...>{}, magnitude<T2...>{});
593 } else {
594 // When the bases are equal, pick whichever has the lower power.
595 constexpr auto common_tail = common_magnitude(magnitude<T1...>{}, magnitude<T2...>{});
596 if constexpr (H1.power < H2.power) {
597 return magnitude<H1>{} * common_tail;
598 } else {
599 return magnitude<H2>{} * common_tail;
600 }
601 }

617// Helper to perform prime factorization at compile time.
618template<std::intmax_t N>
619 requires(N > 0)
620struct prime_factorization {
621 static constexpr std::intmax_t get_or_compute_first_factor()
622 {
623 if constexpr (known_first_factor<N>.has_value()) {
624 return known_first_factor<N>.value();
625 } else {
626 return static_cast<std::intmax_t>(factorizer::find_first_factor(N));
627 }
628 }
629
630 static constexpr std::intmax_t first_base = get_or_compute_first_factor();
631 static constexpr std::intmax_t first_power = multiplicity(first_base, N);
632 static constexpr std::intmax_t remainder = remove_power(first_base, first_power, N);
633
634 static constexpr auto value =
635 magnitude<base_power{first_base, first_power}>{} * prime_factorization<remainder>::value;
636};

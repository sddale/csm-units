52template<Quantity Q>
53inline constexpr Magnitude auto quantity_magnitude = decltype(Q::reference)::mag;

55template<typename QFrom, typename QTo>
56inline constexpr Magnitude auto cast_magnitude = [] {
57 using FromU = TYPENAME QFrom::unit;
58 using ToU = TYPENAME QTo::unit;
59 if constexpr (same_unit_reference<FromU, ToU>::value) {
60 return FromU::mag / ToU::mag;
61 } else {
62 return quantity_magnitude<QFrom> / quantity_magnitude<QTo>;
63 }
64}();

67struct cast_traits;

69template<typename From, typename To>
70 requires common_type_with_<std::common_type_t<From, To>, std::intmax_t>
71struct cast_traits<From, To> {
72 using ratio_type = std::common_type_t<std::common_type_t<From, To>, std::intmax_t>;
73 using rep_type = ratio_type;
74};

76template<typename From, typename To>
77 requires(!common_type_with_<std::common_type_t<From, To>, std::intmax_t> &&
78 scalable_number_<std::common_type_t<From, To>, std::intmax_t> &&
79 requires { typename std::common_type_t<From, To>::value_type; } &&
80 common_type_with_<typename std::common_type_t<From, To>::value_type, std::intmax_t>)
81struct cast_traits<From, To> {
82 using ratio_type = std::common_type_t<typename std::common_type_t<From, To>::value_type, std::intmax_t>;
83 using rep_type = std::common_type_t<From, To>;
84};

100template<Quantity To, typename D, typename U, scalable_with_<typename To::rep> Rep>
101 requires QuantityOf<To, D> && (std::constructible_from<typename To::rep, std::common_type_t<typename To::rep, Rep>>)
102[[nodiscard]] constexpr auto quantity_cast(const quantity<D, U, Rep>& q)
103{
104 using traits = detail::cast_traits<Rep, typename To::rep>;
105 using ratio_type = TYPENAME traits::ratio_type;
106 using rep_type = TYPENAME traits::rep_type;
107
108 constexpr Magnitude auto c_mag = detail::cast_magnitude<quantity<D, U, Rep>, To>;
109 constexpr Magnitude auto num = numerator(c_mag);
110 constexpr Magnitude auto den = denominator(c_mag);
111 constexpr Magnitude auto irr = c_mag * (den / num);
112
113 constexpr auto val = [](Magnitude auto m) { return get_value<ratio_type>(m); };
114 return To(static_cast<TYPENAME To::rep>(static_cast<rep_type>(q.number()) * val(num) / val(den) * val(irr)));
115}

268template<typename CastSpec, typename K, typename U, typename Rep>
269[[nodiscard]] constexpr QuantityKind auto quantity_kind_cast(const quantity_kind<K, U, Rep>& qk)
270 requires requires {
271 requires is_specialization_of<CastSpec, quantity_kind>;
272 requires requires { quantity_cast<typename CastSpec::quantity_type>(qk.common()); };
273 } || requires {
274 requires Kind<CastSpec>;
275 requires UnitOf<U, typename CastSpec::dimension>;
276 } || requires { quantity_cast<CastSpec>(qk.common()); } // TODO: Simplify when Clang catches up.
277{
278 if constexpr (is_specialization_of<CastSpec, quantity_kind>)
279 return CastSpec(quantity_cast<typename CastSpec::quantity_type>(qk.common()));
280 else if constexpr (Kind<CastSpec>)
281 return quantity_kind<CastSpec, U, Rep>(qk.common());
282 else {
283 auto q{quantity_cast<CastSpec>(qk.common())};
284 using Q = decltype(q);
285 return quantity_kind<K, typename Q::unit, typename Q::rep>(static_cast<Q&&>(q));
286 }

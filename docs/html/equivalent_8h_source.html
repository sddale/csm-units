80template<typename T, typename U>
81concept EquivalentPointOrigins =
82 RebindablePointOriginFor<T, typename U::dimension> && RebindablePointOriginFor<U, typename T::dimension> &&
83 std::same_as<T, rebind_point_origin_dimension<U, typename T::dimension>> &&
84 std::same_as<U, rebind_point_origin_dimension<T, typename U::dimension>>;

86template<PointOrigin T, PointOrigin U>
87struct equivalent_impl<T, U> :

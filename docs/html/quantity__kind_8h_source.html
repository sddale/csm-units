39template<Kind K>
40inline constexpr auto make_quantity_kind_fn = [](auto&& q) {
41 using Q = std::remove_reference_t<decltype(q)>;
42 return quantity_kind<K, typename Q::unit, typename Q::rep>(std::forward<decltype(q)>(q));
43};

45template<QuantityKind QK>
46inline constexpr auto& make_quantity_kind = make_quantity_kind_fn<typename QK::kind_type>;

48template<Kind K>
49inline constexpr auto downcasted_kind_fn = [](auto q) {
50 using Q = decltype(q);
51 return make_quantity_kind_fn<downcast_kind<K, typename Q::dimension>>(std::move(q));

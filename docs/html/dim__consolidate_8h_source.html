57template<BaseDimension Dim, std::intmax_t Num1, std::intmax_t Den1, std::intmax_t Num2, std::intmax_t Den2,
58 typename... ERest>
59struct dim_consolidate<exponent_list<exponent<Dim, Num1, Den1>, exponent<Dim, Num2, Den2>, ERest...>> {
60 using r1 = std::ratio<Num1, Den1>;
61 using r2 = std::ratio<Num2, Den2>;
62 using r = std::ratio_add<r1, r2>;
63 using type = conditional<r::num == 0, typename dim_consolidate<exponent_list<ERest...>>::type,
64 typename dim_consolidate<exponent_list<exponent<Dim, r::num, r::den>, ERest...>>::type>;
65};

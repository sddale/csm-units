51template<std::intmax_t Num, std::intmax_t Den = 1, Quantity Q>
52 requires detail::non_zero<Den>
53[[nodiscard]] inline auto pow(const Q& q) noexcept
54 requires requires { pow(q.number(), 1.0); } || requires { std::pow(q.number(), 1.0); }
55{
56 using rep = TYPENAME Q::rep;
57 if constexpr (Num == 0) {
58 return rep(1);
59 } else {
60 using dim = dimension_pow<typename Q::dimension, Num, Den>;
61 using unit = downcast_unit<dim, pow<ratio{Num, Den}>(Q::reference.mag) / dim::mag>;
62 using std::pow;
63 return quantity<dim, unit, rep>(
64 static_cast<rep>(pow(q.number(), static_cast<double>(Num) / static_cast<double>(Den))));
65 }
66}

157template<Unit To, typename D, typename U, typename Rep>
158[[nodiscard]] constexpr quantity<D, To, Rep> floor(const quantity<D, U, Rep>& q) noexcept
159 requires((!treat_as_floating_point<Rep>) || requires { floor(q.number()); } ||
160 requires { std::floor(q.number()); }) &&
161 (std::same_as<To, U> || requires {
162 ::units::quantity_cast<To>(q);
163 quantity<D, To, Rep>::one();
164 })
165{
166 const auto handle_signed_results = [&]<typename T>(const T& res) {
167 if (res > q) {
168 return res - T::one();
169 }
170 return res;
171 };
172 if constexpr (treat_as_floating_point<Rep>) {
173 using std::floor;
174 if constexpr (std::is_same_v<To, U>) {
175 return quantity<D, To, Rep>(floor(q.number()));
176 } else {
177 return handle_signed_results(quantity<D, To, Rep>(floor(quantity_cast<To>(q).number())));
178 }
179 } else {
180 if constexpr (std::is_same_v<To, U>) {
181 return q;
182 } else {
183 return handle_signed_results(quantity_cast<To>(q));
184 }
185 }
186}

258template<Unit To, typename D, typename U, typename Rep>
259[[nodiscard]] constexpr quantity<D, To, Rep> round(const quantity<D, U, Rep>& q) noexcept
260 requires((!treat_as_floating_point<Rep>) || requires { round(q.number()); } ||
261 requires { std::round(q.number()); }) &&
262 (std::same_as<To, U> || requires {
263 ::units::floor<To>(q);
264 quantity<D, To, Rep>::one();
265 })
266{
267 if constexpr (std::is_same_v<To, U>) {
268 if constexpr (treat_as_floating_point<Rep>) {
269 using std::round;
270 return quantity<D, To, Rep>(round(q.number()));
271 } else {
272 return q;
273 }
274 } else {
275 const auto res_low = units::floor<To>(q);
276 const auto res_high = res_low + decltype(res_low)::one();
277 const auto diff0 = q - res_low;
278 const auto diff1 = res_high - q;
279 if (diff0 == diff1) {
280 if (static_cast<int>(res_low.number()) & 1) {
281 return res_high;
282 }
283 return res_low;
284 }
285 if (diff0 < diff1) {
286 return res_low;
287 }
288 return res_high;
289 }
290}

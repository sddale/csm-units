42// Computes the rational gcd of n1/d1 and n2/d2
43[[nodiscard]] constexpr auto gcd_frac(std::intmax_t n1, std::intmax_t d1, std::intmax_t n2, std::intmax_t d2) noexcept
44{
45 // Short cut for equal ratios
46 if (n1 == n2 && d1 == d2) {
47 return std::array{n1, d1};
48 }
49
50 // gcd(a/b,c/d) = gcd(a⋅d, c⋅b) / b⋅d
51
52 assert(std::numeric_limits<std::intmax_t>::max() / n1 > d2);
53 assert(std::numeric_limits<std::intmax_t>::max() / n2 > d1);
54
55 std::intmax_t num = std::gcd(n1 * d2, n2 * d1);
56
57 assert(std::numeric_limits<std::intmax_t>::max() / d1 > d2);
58
59 std::intmax_t den = d1 * d2;
60
61 std::intmax_t gcd = std::gcd(num, den);
62
63 return std::array{num / gcd, den / gcd};
64}

66constexpr void normalize(std::intmax_t& num, std::intmax_t& den)
67{
68 if (num == 0) {
69 den = 1;
70 return;
71 }
72
73 std::intmax_t gcd = std::gcd(num, den);
74 num = num * (den < 0 ? -1 : 1) / gcd;
75 den = detail::abs(den) / gcd;
76}

78[[nodiscard]] constexpr std::intmax_t safe_multiply(std::intmax_t lhs, std::intmax_t rhs)
79{
80 constexpr std::intmax_t c = std::uintmax_t(1) << (sizeof(std::intmax_t) * 4);
81
82 const std::intmax_t a0 = detail::abs(lhs) % c;
83 const std::intmax_t a1 = detail::abs(lhs) / c;
84 const std::intmax_t b0 = detail::abs(rhs) % c;
85 const std::intmax_t b1 = detail::abs(rhs) / c;
86
87 gsl_Expects(a1 == 0 || b1 == 0); // overflow in multiplication
88 gsl_Expects(a0 * b1 + b0 * a1 < (c >> 1)); // overflow in multiplication
89 gsl_Expects(b0 * a0 <= INTMAX_MAX); // overflow in multiplication
90 gsl_Expects((a0 * b1 + b0 * a1) * c <= INTMAX_MAX - b0 * a0); // overflow in multiplication
91
92 return lhs * rhs;
93}

45template<typename CharT, std::size_t N>
46struct basic_fixed_string {
47 CharT data_[N + 1] = {};
48
49 using iterator = CharT*;
50 using const_iterator = const CharT*;
51
52 constexpr explicit(false) basic_fixed_string(CharT ch) noexcept { data_[0] = ch; }
53
54 constexpr explicit(false) basic_fixed_string(const CharT (&txt)[N + 1]) noexcept
55 {
56 if constexpr (N != 0)
57 for (std::size_t i = 0; i < N; ++i) data_[i] = txt[i];
58 }
59
60 [[nodiscard]] constexpr bool empty() const noexcept { return N == 0; }
61 [[nodiscard]] constexpr std::size_t size() const noexcept { return N; }
62 [[nodiscard]] constexpr const CharT* data() const noexcept { return data_; }
63 [[nodiscard]] constexpr const CharT* c_str() const noexcept { return data(); }
64 [[nodiscard]] constexpr const CharT& operator[](std::size_t index) const noexcept { return data()[index]; }
65 [[nodiscard]] constexpr CharT operator[](std::size_t index) noexcept { return data()[index]; }
66
67 [[nodiscard]] constexpr iterator begin() noexcept { return data(); }
68 [[nodiscard]] constexpr const_iterator begin() const noexcept { return data(); }
69 [[nodiscard]] constexpr iterator end() noexcept { return data() + size(); }
70 [[nodiscard]] constexpr const_iterator end() const noexcept { return data() + size(); }
71
72 template<std::size_t N2>
73 [[nodiscard]] constexpr friend basic_fixed_string<CharT, N + N2> operator+(
74 const basic_fixed_string& lhs, const basic_fixed_string<CharT, N2>& rhs) noexcept
75 {
76 CharT txt[N + N2 + 1] = {};
77
78 for (size_t i = 0; i != N; ++i) txt[i] = lhs[i];
79 for (size_t i = 0; i != N2; ++i) txt[N + i] = rhs[i];
80
81 return basic_fixed_string<CharT, N + N2>(txt);
82 }
83
84 [[nodiscard]] constexpr bool operator==(const basic_fixed_string& other) const
85 {
86 if (size() != other.size()) return false;
87 return detail::equal(begin(), end(), other.begin()); // TODO std::ranges::equal(*this, other)
88 }
89
90 template<std::size_t N2>
91 [[nodiscard]] friend constexpr bool operator==(const basic_fixed_string&, const basic_fixed_string<CharT, N2>&)
92 {
93 return false;
94 }
95
96 template<std::size_t N2>
97 [[nodiscard]] friend constexpr auto operator<=>(const basic_fixed_string& lhs,
98 const basic_fixed_string<CharT, N2>& rhs)
99 {
100 // TODO std::lexicographical_compare_three_way(lhs.begin(), lhs.end(), rhs.begin(), rhs.end());
101 return detail::lexicographical_compare_three_way(lhs.begin(), lhs.end(), rhs.begin(), rhs.end());
102 }
103};

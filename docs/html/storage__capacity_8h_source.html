74struct dim_storage_capacity : base_dimension<"M", byte> {};

76template<typename T>
77concept StorageCapacity = QuantityOf<T, dim_storage_capacity>;
78
79template<UnitOf<dim_storage_capacity> U, Representation Rep = double>
80using storage_capacity = quantity<dim_storage_capacity, U, Rep>;

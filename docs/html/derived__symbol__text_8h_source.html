32template<bool Divide, std::size_t NegativeExpCount, std::size_t Idx>
33constexpr auto operator_text()
34{
35 if constexpr (Idx == 0) {
36 if constexpr (Divide && NegativeExpCount == 1) {
37 return basic_fixed_string("1/");
38 } else {
39 return basic_fixed_string("");
40 }
41 } else {
42 if constexpr (Divide && NegativeExpCount == 1) {
43 return basic_fixed_string("/");
44 } else {
45 return basic_symbol_text("⋅", " ");
46 }
47 }
48}

50template<typename E, basic_symbol_text Symbol, std::size_t NegativeExpCount, std::size_t Idx>
51constexpr auto exp_text()
52{
53 // get calculation operator + symbol
54 const auto txt = operator_text<(E::num < 0), NegativeExpCount, Idx>() + Symbol;
55 if constexpr (E::den != 1) {
56 // add root part
57 return txt + basic_fixed_string("^(") + regular<abs(E::num)>() + basic_fixed_string("/") + regular<E::den>() +
58 basic_fixed_string(")");
59 } else if constexpr (E::num != 1) {
60 // add exponent part
61 if constexpr (NegativeExpCount > 1) { // no '/' sign here (only negative exponents)
62 return txt + superscript<E::num>();
63 } else if constexpr (E::num != -1) { // -1 is replaced with '/' sign here
64 return txt + superscript<abs(E::num)>();
65 } else {
66 return txt;
67 }
68 } else {
69 return txt;
70 }
71}

73template<typename... Es>
74inline constexpr int negative_exp_count = ((Es::num < 0 ? 1 : 0) + ... + 0);
75
76template<typename... Us, typename... Es, std::size_t... Idxs>
77constexpr auto derived_symbol_text(exponent_list<Es...>, std::index_sequence<Idxs...>)
78{
79 constexpr auto neg_exp = negative_exp_count<Es...>;
80 return (exp_text<Es, Us::symbol, neg_exp, Idxs>() + ...);
81}

83template<DerivedDimension Dim, Unit... Us>
84constexpr auto derived_symbol_text()
85{
86 return derived_symbol_text<Us...>(typename Dim::recipe(), std::index_sequence_for<Us...>());
87}

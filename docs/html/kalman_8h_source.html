52// state
53template<QuantityOrQuantityPoint... QQPs>
54 requires(sizeof...(QQPs) > 0) && (sizeof...(QQPs) <= 3) && are_time_derivatives<QQPs...>
55struct state {

57 constexpr state(QQPs... qqps) : variables_(std::move(qqps)...) {}

94// kalman gain
95template<units::Quantity Q>
96constexpr units::dimensionless<units::one> kalman_gain(Q estimate_uncertainty, Q measurement_uncertainty)
97{
98 return estimate_uncertainty / (estimate_uncertainty + measurement_uncertainty);
99}

101// state update
102template<typename Q, QuantityOrQuantityPoint QM, units::Dimensionless K>
103 requires units::equivalent<typename Q::dimension, typename QM::dimension>
104constexpr state<Q> state_update(const state<Q>& predicted, QM measured, K gain)
105{
106 return {get<0>(predicted) + gain * (measured - get<0>(predicted))};
107}
108
109template<typename Q1, typename Q2, QuantityOrQuantityPoint QM, units::Dimensionless K, units::isq::Time T>
110 requires units::equivalent<typename Q1::dimension, typename QM::dimension>
111constexpr state<Q1, Q2> state_update(const state<Q1, Q2>& predicted, QM measured, std::array<K, 2> gain, T interval)
112{
113 const auto q1 = get<0>(predicted) + get<0>(gain) * (measured - get<0>(predicted));
114 const auto q2 = get<1>(predicted) + get<1>(gain) * (measured - get<0>(predicted)) / interval;
115 return {q1, q2};
116}

118template<typename Q1, typename Q2, typename Q3, QuantityOrQuantityPoint QM, units::Dimensionless K, units::isq::Time T>
119 requires units::equivalent<typename Q1::dimension, typename QM::dimension>
120constexpr state<Q1, Q2, Q3> state_update(const state<Q1, Q2, Q3>& predicted, QM measured, std::array<K, 3> gain,
121 T interval)
122{
123 const auto q1 = get<0>(predicted) + get<0>(gain) * (measured - get<0>(predicted));
124 const auto q2 = get<1>(predicted) + get<1>(gain) * (measured - get<0>(predicted)) / interval;
125 const auto q3 = get<2>(predicted) + get<2>(gain) * (measured - get<0>(predicted)) / (interval * interval / 2);
126 return {q1, q2, q3};
127}

129// covariance update
130template<units::Quantity Q, units::Dimensionless K>
131constexpr Q covariance_update(Q uncertainty, K gain)
132{
133 return (1 - gain) * uncertainty;

136// state extrapolation
137template<typename Q1, typename Q2, units::isq::Time T>
138constexpr state<Q1, Q2> state_extrapolation(const state<Q1, Q2>& estimated, T interval)
139{
140 const auto q1 = get<0>(estimated) + get<1>(estimated) * interval;
141 const auto q2 = get<1>(estimated);
142 return {q1, q2};
143}

145template<typename Q1, typename Q2, typename Q3, units::isq::Time T>
146constexpr state<Q1, Q2, Q3> state_extrapolation(const state<Q1, Q2, Q3>& estimated, T interval)
147{
148 const auto q1 = get<0>(estimated) + get<1>(estimated) * interval + get<2>(estimated) * pow<2>(interval) / 2;
149 const auto q2 = get<1>(estimated) + get<2>(estimated) * interval;
150 const auto q3 = get<2>(estimated);
151 return {q1, q2, q3};
152}

171 template<typename FormatContext>
172 auto format(const kalman::state<Qs...>& s, FormatContext& ctx)
173 {
174 std::string value_buffer;
175 auto to_value_buffer = std::back_inserter(value_buffer);
176 if (specs.precision != -1) {
177 if constexpr (sizeof...(Qs) == 1)
178 UNITS_STD_FMT::format_to(to_value_buffer, "{1:%.{0}Q %q}", specs.precision, kalman::get<0>(s));
179 else if constexpr (sizeof...(Qs) == 2)
180 UNITS_STD_FMT::format_to(to_value_buffer, "{{ {1:%.{0}Q %q}, {2:%.{0}Q %q} }}", specs.precision,
181 kalman::get<0>(s), kalman::get<1>(s));
182 else
183 UNITS_STD_FMT::format_to(to_value_buffer, "{{ {1:%.{0}Q %q}, {2:%.{0}Q %q}, {3:%.{0}Q %q} }}", specs.precision,
184 kalman::get<0>(s), kalman::get<1>(s), kalman::get<2>(s));
185 } else {
186 if constexpr (sizeof...(Qs) == 1)
187 UNITS_STD_FMT::format_to(to_value_buffer, "{}", kalman::get<0>(s));
188 else if constexpr (sizeof...(Qs) == 2)
189 UNITS_STD_FMT::format_to(to_value_buffer, "{{ {}, {} }}", kalman::get<0>(s), kalman::get<1>(s));
190 else
191 UNITS_STD_FMT::format_to(to_value_buffer, "{{ {}, {}, {} }}", kalman::get<0>(s), kalman::get<1>(s),
192 kalman::get<2>(s));
193 }
194
195 std::string global_format_buffer;
196 units::detail::quantity_global_format_specs<char> global_specs = {specs.fill, specs.align, specs.width};
197 units::detail::format_global_buffer(std::back_inserter(global_format_buffer), global_specs);
198
199 return UNITS_STD_FMT::vformat_to(ctx.out(), global_format_buffer, UNITS_STD_FMT::make_format_args(value_buffer));
200 }
201private:
202 units::detail::dynamic_format_specs<char> specs;
203};

215 {
216 units::Quantity auto q = [](const Q& t) {
217 if constexpr (units::Quantity<Q>)
218 return t;
219 else
220 return t.relative();
221 }(kalman::get<0>(e.state));
222
223 std::string value_buffer;

225 if (specs.precision != -1) {

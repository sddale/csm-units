35constexpr bool is_prime_by_trial_division(std::uintmax_t n)
36{
37 for (std::uintmax_t f = 2; f * f <= n; f += 1 + (f % 2)) {
38 if (n % f == 0) {
39 return false;
40 }
41 }
42 return true;
43}

45// Return the first factor of n, as long as it is either k or n.
46//
47// Precondition: no integer smaller than k evenly divides n.
48constexpr std::optional<std::uintmax_t> first_factor_maybe(std::uintmax_t n, std::uintmax_t k)
49{
50 if (n % k == 0) {
51 return k;
52 }
53 if (k * k > n) {
54 return n;
55 }
56 return std::nullopt;
57}

59template<std::size_t N>
60constexpr std::array<std::uintmax_t, N> first_n_primes()
61{
62 std::array<std::uintmax_t, N> primes;
63 primes[0] = 2;
64 for (std::size_t i = 1; i < N; ++i) {
65 primes[i] = primes[i - 1] + 1;
66 while (!is_prime_by_trial_division(primes[i])) {
67 ++primes[i];
68 }
69 }
70 return primes;
71}

73template<std::size_t N, typename Callable>
74constexpr void call_for_coprimes_up_to(std::uintmax_t n, const std::array<std::uintmax_t, N>& basis, Callable&& call)
75{
76 for (std::uintmax_t i = 0u; i < n; ++i) {
77 if (std::apply([&i](auto... primes) { return ((i % primes != 0) && ...); }, basis)) {
78 call(i);
79 }
80 }
81}

83template<std::size_t N>
84constexpr std::size_t num_coprimes_up_to(std::uintmax_t n, const std::array<std::uintmax_t, N>& basis)
85{
86 std::size_t count = 0u;
87 call_for_coprimes_up_to(n, basis, [&count](auto) { ++count; });
88 return count;
89}

91template<std::size_t ResultSize, std::size_t N>
92constexpr auto coprimes_up_to(std::size_t n, const std::array<std::uintmax_t, N>& basis)
93{
94 std::array<std::uintmax_t, ResultSize> coprimes;
95 std::size_t i = 0u;
96
97 call_for_coprimes_up_to(n, basis, [&coprimes, &i](std::uintmax_t cp) { coprimes[i++] = cp; });
98
99 return coprimes;
100}

112// A configurable instantiation of the "wheel factorization" algorithm [1] for prime numbers.
113//
114// Instantiate with N to use a "basis" of the first N prime numbers. Higher values of N use fewer trial divisions, at
115// the cost of additional space. The amount of space consumed is roughly the total number of numbers that are a) less
116// than the _product_ of the basis elements (first N primes), and b) coprime with every element of the basis. This
117// means it grows rapidly with N. Consider this approximate chart:
118//
119// N | Num coprimes | Trial divisions needed
120// --+--------------+-----------------------
121// 1 | 1 | 50.0 %
122// 2 | 2 | 33.3 %
123// 3 | 8 | 26.7 %
124// 4 | 48 | 22.9 %
125// 5 | 480 | 20.8 %
126//
127// Note the diminishing returns, and the rapidly escalating costs. Consider this behaviour when choosing the value of N
128// most appropriate for your needs.
129//
130// [1] https://en.wikipedia.org/wiki/Wheel_factorization
131template<std::size_t BasisSize>
132struct wheel_factorizer {
133 static constexpr auto basis = first_n_primes<BasisSize>();
134 static constexpr auto wheel_size = product(basis);
135 static constexpr auto coprimes_in_first_wheel =
136 coprimes_up_to<num_coprimes_up_to(wheel_size, basis)>(wheel_size, basis);
137
138 static constexpr std::uintmax_t find_first_factor(std::uintmax_t n)
139 {
140 if (const auto k = detail::get_first_of(basis, [&](auto p) { return first_factor_maybe(n, p); })) return *k;
141
142 if (const auto k = detail::get_first_of(std::next(begin(coprimes_in_first_wheel)), end(coprimes_in_first_wheel),
143 [&](auto p) { return first_factor_maybe(n, p); }))
144 return *k;
145
146 for (std::size_t wheel = wheel_size; wheel < n; wheel += wheel_size)
147 if (const auto k =
148 detail::get_first_of(coprimes_in_first_wheel, [&](auto p) { return first_factor_maybe(n, wheel + p); }))
149 return *k;
150 return n;
151 }
152
153 static constexpr bool is_prime(std::size_t n) { return (n > 1) && find_first_factor(n) == n; }
154};
